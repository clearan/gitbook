如何中断promise

如何中断promise

const p = new Promise((resolve, reject) => {
    resolve(1)
})
p.then(data => {
    console.log(data) // 1
}, reason => {
    console.log(reason)
})

// 下面不让实例对象p调用then方法,p处于pending状态，没有被推向已决阶段，则不会走到回调函数then里面
const p = new Promise((resolve, reject) => {})
p.then(data => {
    console.log(data) // 1
}, reason => {
    console.log(reason)
})

// 同理，如何不让第一个例子p实例继续调用then呢
const p = new Promise((resolve, reject) => {
    resolve(1)
})
p.then(data => {
    console.log(data) // 1
    return new Promise(() => {}) // 返回一个promise，执行函数为空函数，砍掉未决推向已决。
}, reason => {
    console.log(reason)
})
